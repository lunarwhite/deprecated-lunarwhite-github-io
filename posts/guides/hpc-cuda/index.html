1.4 简单的数组一一对应相加«<1,1»>

#include <cuda_runtime.h>
#include <stdio.h>
#include "device_launch_parameters.h"

__global__ void add(int* a, int* b, int* c){
    *c = *a + *b;
}

int main(){
    int a, b, c; // host copy
    int *d_a, *d_b, *d_c; // device copy
    int size = sizeof(int);

    // allocate for device copy
    cudaMalloc((void **)&d_a, size);
    cudaMalloc((void **)&d_b, size);
    cudaMalloc((void **)&d_c, size);

    // setup input values
    a = 2;
    b = 7;

    // copy inputs to device
    cudaMemcpy(d_a, &a, size, cudaMemcpyHostToDevice);
    cudaMemcpy(d_b, &b, size, cudaMemcpyHostToDevice);

    // launch add() kernel on GPU
    add<<<1,1>>>(d_a, d_b, d_c);

    // copy result back to host
    cudaMemcpy(&c, d_c, size, cudaMemcpyDeviceToHost) ;

    printf("c = %d\n", c);

    // cleanup
    cudaFree(d_a);
    cudaFree(d_b);
    cudaFree(d_c);

    return 0;
}

#include <cuda_runtime.h>
#include <stdio.h>
#include <stdlib.h>

#define N 512

__global__ void add(int* a, int* b, int* c) {
    c[blockIdx.x] = a[blockIdx.x] + b[blockIdx.x];
}

void random_ints(int* a, int NN) { // 返回一个0~10000的数
    int i;
    for(i = 0; i < NN; ++i)
    a[i] = rand()%10000;
}

int main(void) {
    int *a, *b, *c;
    int *d_a, *d_b, *d_c;

    // alloc for device

    cudaMalloc((void**)&d_b, size);
    cudaMalloc((void**)&d_c, size);

    // alloc for host
    a = (int *)malloc(size);
    random_ints(a, N);
    b = (int *)malloc(size);
    random_ints(b, N);
    c = (int *)malloc(size);

    cudaMemcpy(d_b, b, size, cudaMemcpyHostToDevice);

    add<<<N,1>>>(d_a, d_b, d_c);

    // copy result back to host
    cudaMemcpy(c, d_c, size, cudaMemcpyDeviceToHost);

    for(int i = 0;i < N;i++)
        printf("c[%d] = %d\n", i, c[i]);

    // cleanup
    free(a);
    free(b);
    free(c);
    cudaFree(d_a);
    cudaFree(d_b);
    cudaFree(d_c);

    return 0;
}

1.7 block和threads结合使用

#include <cuda_runtime.h>
#include <stdio.h>
#include <stdlib.h>
#include "device_launch_parameters.h"
#define N (2048*2048)
#define THREADS_PER_BLOCK 512

__global__ void add(int *a, int *b, int *c) {
    int index = threadIdx.x + blockIdx.x * blockDim.x;
    c[index] = a[index] + b[index];
}

void random_ints(int *a, int NN) {
    int i;
    for(i = 0; i < NN; ++i)
        a[i] = rand()%10000;
}

int main(void) {

#include <stdio.h>

void add_cpu(float *a, float *b, float *c, int NN){
    int idx = 0;

        c[idx] = a[idx] + b[idx];
}

int main(){
    float a[N]={1, 2, 3, 4};
    float b[N]={4, 3, 2, 1};
    float c[N]={0};
    add_cpu(&a, &b, &c, N);

    int i = 0;
    for(i = 0;i < N;i++)
        printf("c[%d] = %f\n", i, c[i]);

    return 0;
}

2.1 纯 CUDA 加法

#include <cuda_runtime.h>
#include <stdio.h>
#include "device_launch_parameters.h"

#define N 4

__global__ void add_gpu(float *a, float *b, int NN) {
    int idx =blockIdx.x* blockDim.x+ threadIdx.x;
    if (idx < NN)
        a[idx] += b[idx];
}

void random_ints(float *a, int NN) {
    int i;
    for (i = 0; i < NN; ++i)
        a[i] = rand() / (double)(RAND_MAX / 100);
}

int main() {
    float *a, *b;
    float *d_a, *d_b;
    int size = N * sizeof(int);

    // alloc for device
    cudaMalloc((void**)&d_a, size);
    cudaMalloc((void**)&d_b, size);

    // alloc for host
    a = (float *)malloc(size); random_ints(a, N);
    b = (float *)malloc(size); random_ints(b, N);

    for (int i = 0; i < N; i++) {
        printf("a[%d] = %f", i, a[i]);
        printf("\tb[%d] = %f\n", i, b[i]);
    }

    // copy inputs to device
    cudaMemcpy(d_a, a, size, cudaMemcpyHostToDevice);
    cudaMemcpy(d_b, b, size, cudaMemcpyHostToDevice);

    dim3 dimBlock (512);
    dim3 dimGrid( ceil( N / 512 ));

    // launch add() kernel on GPU
    add_gpu<<<dimGrid, dimBlock>>>(d_a, d_b, N);

    // copy result back to host
    cudaMemcpy(a, d_a, size, cudaMemcpyDeviceToHost);

    for (int i = 0; i < N; i++) {
        printf("a[%d] = %f\n", i, a[i]);
    }

    // cleanup
    free(a);
    free(b);
    cudaFree(d_a);
    cudaFree(d_b);

    return 0;
}

2.2 unified-memory

#include <cuda_runtime.h>
#include <stdio.h>
#include "device_launch_parameters.h"

#define N 10*10

__global__ void AplusB(int *ret, int a, int b) {
    ret[threadIdx.x] = a + b + threadIdx.x;
}

int main(){
    int *ret;
    cudaMallocManaged(&ret, N*sizeof(int));

    AplusB<<<1,N>>>(ret, 10, 100);

    cudaDeviceSynchronize();

    for(int i=0; i<N; i++)
        printf("%d: A+B= %d\n", i, ret[i]);

    cudaFree(ret);
    return 0;
}

2.4 矩阵乘法

#include "cuda_runtime.h"
#include "device_launch_parameters.h"
#include <stdio.h>
#define Row  126
#define Col 250

__global__ void matrix_mul_gpu(int *M, int* N, int* P, int width) {
    int i = threadIdx.x + blockDim.x * blockIdx.x;
    int j = threadIdx.y + blockDim.y * blockIdx.y;

    int sum = 0;
    for(int k=0;k<width;k++) {
        int a = M[j*width+k];
        int b = N[k*width+i];
        sum += a*b;
    }
    P[j*width+i] = sum;
}

int main(){
    int *d_dataA, *d_dataB, *d_dataC;

    // alloc for host
    int *A = (int *)malloc(sizeof(int) * Row * Col);
    int *B = (int *)malloc(sizeof(int) * Row * Col);
    int *C = (int *)malloc(sizeof(int) * Row * Col);

    // alloc for device
    cudaMalloc((void**)&d_dataA, sizeof(int) *Row*Col);
    cudaMalloc((void**)&d_dataB, sizeof(int) *Row*Col);
    cudaMalloc((void**)&d_dataC, sizeof(int) *Row*Col);

    for (int i = 0; i < Row*Col; i++) {
        A[i] = 90;
        B[i] = 10;
    }

    // copy inputs to device
    cudaMemcpy(d_dataA, A, sizeof(int) * Row * Col, cudaMemcpyHostToDevice);
    cudaMemcpy(d_dataB, B, sizeof(int) * Row * Col, cudaMemcpyHostToDevice);

    dim3 threadPerBlock(16, 16);
    dim3 blockNumber((Col+threadPerBlock.x-1)/ threadPerBlock.x, (Row+threadPerBlock.y-1)/ threadPerBlock.y );

    printf("Block(%d,%d)   Grid(%d,%d).\n", threadPerBlock.x, threadPerBlock.y, blockNumber.x, blockNumber.y);
    matrix_mul_gpu << <blockNumber, threadPerBlock >> > (d_dataA, d_dataB, d_dataC, Col);

    // copy result back to host
    cudaMemcpy(C, d_dataC, sizeof(int) * Row * Col, cudaMemcpyDeviceToHost);

    // cleanup
    free(A);
    free(B);
    free(C);
    cudaFree(d_dataA);
    cudaFree(d_dataB);
    cudaFree(d_dataC);

    return 0;
}
