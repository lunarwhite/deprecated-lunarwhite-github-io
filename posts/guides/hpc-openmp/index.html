1 初识与OpenMP

1.1 OpenMP基本结构

#include <stdlib.h>
#include <omp.h>
#include <stdio.h>
#define NUM_THREADS 8 //设置线程数目为8

int main(){
    //--串行区，由一个初始线程执行
    static long num_steps=1000000000;
    double step;
    int i;
    double x,pi,sum=0.0;

    step=1.0/(double) num_steps;

    //调用运行时库函数，为后面的并行区设置线程数目
    omp_set_num_threads(NUM_THREADS);

    //编译指导语句，创建一个并行区。在该并行区中多线程执行
    #pragma omp parallel for reduction(+:sum) private(x)

    //--并行区，该区域由8个线程并行执行
    for(i=0;i<num_steps;i++){
        x=(i+0.5)*step;
        sum+=4.0/(1.0+x*x);
    }

    //--串行区，只有一个主线程执行
    pi=step*sum;
    printf("Pi=%21.20f(%ld steps)\n",pi,num_steps);
    return 0;
}

1.2 循环并行化

#include <stdlib.h>
#include <omp.h>
#include <stdio.h>
#define NUM_THREADS 8 //设置线程数目为8
#define N 100

    int main(){
    //--串行区，由一个初始线程执行
    int i;
    int x[N];
    int y[N];
    int z[N];

    //初始化x[i] y[i]
    for(i=0; i<N; i++){
        x[i]=i;
        y[i]=i*i;
    }

    //调用运行时库函数，为后面的并行区设置线程数目
    omp_set_num_threads(NUM_THREADS);

    //编译指导语句，创建一个并行区。在该并行区中多线程执行
    #pragma omp parallel for
    for(i=0; i<N; i++){ //--并行区，该区域由8个线程并行执行
        z[i] = x[i]+y[i];
    }

    //--串行区，只有一个主线程执行
    for(i=0; i<N; i++){
        printf("z[%d]=x[%d]+y[%d]=%d\n", i, i, i, z[i]);
    }
    return 0;
}

1.3 私有变量的初始化和终结操作

#include <stdlib.h>
#include <stdio.h>
#include <omp.h>
#define NUM_THREADS 8 //设置线程数目为8
#define N 10

int main(){
    //--串行区

    //初始化
    int val=64;
    int i=0;

    //设置线程数目
    omp_set_num_threads(NUM_THREADS);

    //创建并行区
    #pragma omp parallel for firstprivate(val) lastprivate(val)
    for(i=0;i<N;i++){
        printf("i=%d val=%d\n",i,val);
        if(i==1){
            val=10000;
        }
        printf("i=%d val=%d\n",i,val);
    }

    //--串行区
    printf("val=%d\n",val);
    return 0;
}

1.4 循环嵌套

#include <stdlib.h>
#include <stdio.h>
#include <omp.h>
#define NUM_THREADS 8 //设置线程数目为8
#define N 2
#define M 6

int main(){
    //--串行区

    //初始化
    int i=0;
    int j=0;

    //设置线程数目
    omp_set_num_threads(NUM_THREADS);

    //创建并行区
    #pragma omp parallel for private(j)
    for(i=0; i<N; i++){
        for(j=0; j<M; j++){
        printf("i=%d j=%d\n", i, j);
        }
    }

    return 0;
}

1.5 归约操作的并行化

#include <stdlib.h>
#include <stdio.h>
#include <omp.h>
#define NUM_THREADS 8 //设置线程数目为8

int main(){
    //--串行区
    int a1=0;
    int a2=0;
    int a3=0;
    int i=1;

    //设置线程数目
    omp_set_num_threads(NUM_THREADS);

    //无归约操作求解1-100的和，计算结果不确定
    #pragma omp parallel for private(i)
    for(i=1;i<101;i++){
        a1=a1+i;
    }

    //归约操作求解1-100的和，计算结果正确
    #pragma omp parallel for private(i) reduction(+:a)
    for(i=1;i<101;i++){
        a2=a2+i;
    }

    //串行操作
    for(i=1;i<101;i++){
        a3=a3+i;
    }

    //显示结果
    printf("并行无归约操作\t a1=%d\n",a1);
    printf("并行归约操作\t a2=%d\n",a2);
    printf("串行操作\t a3=%d\n",a3);

    return 0;
}

1.6 非迭代结构并行

#include <stdlib.h>
#include <stdio.h>
#include <omp.h>
#define NUM_THREADS 3 //设置线程数目为8

int alpha(){
    return 100;
}

int beta(){
    return 600;
}

int delta(){
    return 400;
}

int plus(int v,int w){
    return v+w;
}

int multiply(int x,int y){
    return x*y;
}

int main(){
    //设置线程数目
    omp_set_num_threads(NUM_THREADS);

    //初始化
    int v, w, y, x, z;

    //首先并行处理得到v，w，然后并行处理得到x，y，
    //最后打印出z
    //sections与sections之间串行执行

    #pragma omp parallel
    {
        #pragma omp sections
        {
            #pragma omp section
            v = alpha();

            #pragma omp section
            w = beta();
        }
        #pragma omp sections
        {
        #pragma omp section
        x = plus(v,w);

        #pragma omp section
        y = delta();
        }
    }

    z = multiply(x, y);

    printf("v=alpha()=%d\n", v);
    printf("w=beta()=%d\n", w);
    printf("y=delta()=%d\n", y);
    printf("x=plus(v, w)=%d\n", x);
    printf("z=multiply(x, y)=%d\n", z);

    return 0;
}
